20  /**
21   * Prepares the tournament properties by reading them from a property file.
22   * 
23   * <p>
24   * Mandatory properties are:
25   * </p>
26   * 
27   * <dl>
28   * <dt>game.class</dt>
29   * <dd>Which {@link Game} implementation the {@link DroomsTournament} should
30   * use. Fully qualified name.</dd>
31   * <dt>players</dt>
32   * <dd>Path to the file containing descriptions of the {@link Player}s that will
33   * participate in the tournament. See {@link PlayerAssembly}. Relative to the
34   * current working directory.</dd>
35   * <dt>playgrounds</dt>
36   * <dd>Comma-separated list of playground on which to play the tournament.</dd>
37   * </dl>
38   * 
39   * <p>
40   * Optional properties are:
41   * </p>
42   * 
43   * <dl>
44   * <dt>runs</dt>
45   * <dd>How many times should each playground be played. Number > 0, default
46   * value is 1.</dd>
47   * <dt>folder.resources</dt>
48   * <dd>Where to load all input files from, relative to the current working
49   * directory. If it doesn't exist, it is created. Default value is
50   * "src/main/resources".</dd>
51   * <dt>folder.target</dt>
52   * <dd>Where to write all output data, relative to the current working
53   * directory. If it doesn't exist, it is created. Default value is
54   * "target/drooms".</dd>
55   * </dl>
56   * 
57   * FIXME document player config file format.
58   */
59  public class TournamentProperties extends CommonProperties {

61      @SuppressWarnings("unchecked")
62      private static Class<? extends Game> getGameImpl(final String id) {
63          try {
64              return (Class<? extends Game>) Class.forName(id);
65          } catch (final ClassNotFoundException e) {
66              throw new IllegalArgumentException("Cannot instantiate game class.", e);
67          }
68      }

70      private static Playground loadPlayground(final File source, final String name) {
71          final File playgroundFile = new File(source, name + ".playground");
72          try (InputStream is = new FileInputStream(playgroundFile)) {
73              return DefaultPlayground.read(name, is);
74          } catch (final IOException e) {
75              throw new IllegalStateException("Cannot read playground file " + playgroundFile, e);
76          }
77      }
78  
79      public static TournamentProperties read(final File f) {
80          return new TournamentProperties(CommonProperties.loadPropertiesFromFile(f));
81      }

83      private final Class<? extends Game> gameClass;
84      private final File resourceFolder;
85      private final File targetFolder;
86      private final Collection<ImmutablePair<Playground, GameProperties>> playgrounds;
87  
88      private final int numberOfRunsPerPlayground;
89  
90      private final Collection<Player> players;
91  
92      private TournamentProperties(final Properties p) {
93          super(p);
94          this.gameClass = TournamentProperties.getGameImpl(this.getMandatoryProperty("game.class"));
95          this.numberOfRunsPerPlayground = Integer.valueOf(this.getOptionalProperty("runs", "1"));
96          // prepare folders
97          this.resourceFolder = new File(this.getOptionalProperty("folder.resources", "src/main/resources"));
98          if (!this.resourceFolder.exists()) {
99              this.resourceFolder.mkdirs();
100         }
101         this.targetFolder = new File(this.getOptionalProperty("folder.target", "target/drooms"));
102         if (!this.targetFolder.exists()) {
103             this.targetFolder.mkdirs();
104         }
105         // prepare a list of players
106         final File playerConfigFile = new File(this.getMandatoryProperty("players"));
107         this.players = Collections.unmodifiableList(new PlayerAssembly(playerConfigFile).assemblePlayers());
108         // parse the playgrounds
109         final Collection<ImmutablePair<Playground, GameProperties>> playgrounds = new ArrayList<>();
110         for (final String playgroundName : this.getMandatoryProperty("playgrounds").split("\\Q,\\E")) {
111             final Playground playground = TournamentProperties.loadPlayground(this.resourceFolder, playgroundName);
112             final GameProperties props = GameProperties.read(new File(this.resourceFolder, playgroundName + ".cfg"));
113             playgrounds.add(new ImmutablePair<Playground, GameProperties>(playground, props));
114         }
115         this.playgrounds = Collections.unmodifiableCollection(playgrounds);
116     }
