15  /**
16   * Prepares the game properties by reading them from a property file.
17   * 
18   * <p>
19   * Mandatory properties are:
20   * </p>
21   * 
22   * <dl>
23   * <dt>collectibles</dt>
24   * <dd>Which {@link CollectibleType}s exist, as a comma-separated list. Each name in this list will then subsequently
25   * require properties 'collectible.expiration.$NAME', 'collectible.price.$NAME' and 'collectible.probability.$NAME'.</dd>
26   * </dl>
27   * 
28   * <p>
29   * Optional properties are:
30   * </p>
31   * 
32   * <dl>
33   * <dt>worm.length.start (defaults to 3)</dt>
34   * <dd>Length of the worm at the start of the game. Actually, initially each worm will only have a length of 1. But as
35   * it first moves its head, the tail of the worm will stay where the head was, until the worm reaches the specified
36   * starting length. From then on, the length will be kept constant and only changed upon collecting an item.</dd>
37   * <dt>worm.max.turns (defaults to 1000)</dt>
38   * <dd>Maximum length of the game, in case more than 1 worm keeps on surviving.</dd>
39   * <dt>worm.max.inactive.turns (defaults to 3)</dt>
40   * <dd>Maximum number of turns of inactivity after which a player may be terminated, if the game decides so.</dd>
41   * <dt>worm.timeout.seconds (defaults to 1)</dt>
42   * <dd>The maximum amount of time that the {@link Player}'s {@link Strategy} has to make a decision on the next movement
43   * of the worm. If it doesn't make it in time, STAY is enforced, potentially leading to the worm being terminated for
44   * inactivity.</dd>
45   * <dt>worm.survival.bonus (defaults to 5)</dt>
46   * <dd>The amount of points that the worm will be awarded upon surviving another worm.</dd>
47   * </dl>
48   */

101         this.strategyTimeoutInSeconds = Integer.valueOf(this.getOptionalProperty("worm.timeout.seconds", "1"));
102         final Collection<CollectibleType> collectibleTypes = new ArrayList<CollectibleType>();
103         for (final String collectibleName : this.getMandatoryProperty("collectibles").split("\\Q,\\E")) {
104             final int expiration = Integer.valueOf(this.getMandatoryProperty("collectible.expiration."
105                     + collectibleName));
106             final int points = Integer.valueOf(this.getMandatoryProperty("collectible.price." + collectibleName));
107             final BigDecimal prob = new BigDecimal(this.getMandatoryProperty("collectible.probability."
108                     + collectibleName));

110         }
