42  /**
43   * Represents a {@link Player}'s {@link Strategy} in action. This class holds
44   * and maintains Drools engine's state for each particular player.
45   * 
46   * <p>
47   * When asked (see {@link #decideNextMove()}), the strategy should make a decision on the next move, based on the
48   * current state of the working memory. This decision should be sent over the provided 'decision' channel. If not sent,
49   * it will default to STAY. See {@link Move} for the various types of decisions.
50   * </p>
51   * <p>
52   * This class enforces the following requirements on the strategies:
53   * </p>
54   * 
55   * <ul>
56   * <li>'gameEvents' entry point must be declared, where the events not directly related to player actions will be sent.
57   * These events are {@link CollectibleAdditionEvent} and {@link CollectibleRemovalEvent}.</li>
58   * <li>'playerEvents' entry point must be declared, where the player-caused events will be sent. These events are
59   * {@link PlayerMoveEvent} and {@link PlayerDeathEvent}.</li>
60   * <li>'rewardEvents' entry point must be declared, where the reward events will be sent. These events are
61   * {@link CollectibleRewardEvent} and {@link SurvivalRewardEvent}.</li>
62   * </ul>
63   * 
64   * <p>
65   * This class provides the following Drools globals, if declared in the strategy:
66   * </p>
67   * 
68   * <ul>
69   * <li>'logger' implementation of the {@link Logger} interface, to use for logging from within the rules.</li>
70   * <li>'tracker' instance of the {@link PathTracker}, to facilitate path-finding in the rules.</li>
71   * </ul>
72   * 
73   * <p>
74   * Your strategies can be validated for all these - just make your tests extend {@link DroomsTestHelper}.
75   * </p>
76   * 
77   * <p>
78   * The working memory will contain instances of the various helper fact types:
79   * </p>
80   * 
81   * <ul>
82   * <li>{@link GameProperty}, many. Will never change or be removed.</li>
83   * <li>{@link CurrentPlayer}, once. Will never change or be removed.</li>
84   * <li>{@link CurrentTurn}, once. Will change with every turn.</li>
85   * <li>{@link Wall}, many. Will remain constant over the whole game.</li>
86   * <li>{@link Worm}, many. Will be added and removed as the worms will move, but never modified.</li>
87   * </ul>
88   * 
89   */

118     public DecisionMaker(final Player p, final PathTracker tracker, final GameProperties properties,
119             final File reportFolder) {
120         this.player = p;
121         this.session = p.constructKnowledgeBase().newStatefulKnowledgeSession(DecisionMaker.getSessionConfiguration(),
122                 null);
123         if (p.auditSession()) {
124             DecisionMaker.LOGGER.info("Auditing the Drools session is enabled.");
125             this.sessionAudit = KnowledgeRuntimeLoggerFactory.newFileLogger(this.session,
126                     reportFolder.getAbsolutePath() + File.separator + "player-" + this.player.getName() + "-session");
127         } else {
128             DecisionMaker.LOGGER.info("Auditing the Drools session is disabled.");
129             this.sessionAudit = null;
130         }
131         // this is where we listen for decisions
132         this.session.registerChannel("decision", this);
133         // validate session
134         final DroomsKnowledgeSessionValidator validator = new DroomsKnowledgeSessionValidator(this.session);
135         if (!validator.isValid()) {
136             throw new IllegalStateException("Player " + this.player.getName() + " has a malformed strategy: "
137                     + validator.getErrors().get(0));
138         }
139         if (!validator.isClean()) {
140             for (final String message : validator.getWarnings()) {
141                 DecisionMaker.LOGGER.info("Player {} has an incomplete strategy: {}", this.player.getName(), message);
142             }
143         }
144         // this is where we will send events from the game
145         this.rewardEvents = this.session.getWorkingMemoryEntryPoint("rewardEvents");
146         this.gameEvents = this.session.getWorkingMemoryEntryPoint("gameEvents");
147         this.playerEvents = this.session.getWorkingMemoryEntryPoint("playerEvents");
148         // configure the globals for the session
149         DecisionMaker.setGlobal(this.session, "tracker", tracker);
150         DecisionMaker.setGlobal(this.session, "logger",
151                 LoggerFactory.getLogger("org.drooms.players." + this.player.getName()));
152         /*
153          * insert playground walls; make sure the playground is always
154          * surrounded with walls.
155          */
156         final Playground playground = tracker.getPlayground();
157         for (int x = -1; x <= playground.getWidth(); x++) {
158             for (int y = -1; y <= playground.getHeight(); y++) {
159                 if (!playground.isAvailable(x, y)) {
160                     this.session.insert(new Wall(Node.getNode(x, y)));
161                 }
162             }
163         }
164         // insert info about the game configuration
165         this.session.insert(new GameProperty(GameProperty.Name.MAX_TURNS, properties.getMaximumTurns()));
166         this.session
167                 .insert(new GameProperty(GameProperty.Name.MAX_INACTIVE_TURNS, properties.getMaximumInactiveTurns()));
168         this.session.insert(new GameProperty(GameProperty.Name.DEAD_WORM_BONUS, properties.getDeadWormBonus()));
169         this.session.insert(new GameProperty(GameProperty.Name.TIMEOUT_IN_SECONDS, properties
170                 .getStrategyTimeoutInSeconds()));
171         // insert info about the game status
172         this.currentTurn = this.session.insert(new CurrentTurn(0));
173         this.session.insert(new CurrentPlayer(p));
174     }

176     /**
177      * Call on the Drools engine to make the decision on worm's next move,
178      * according to the {@link Player}'s {@link Strategy}.
179      * 
180      * @return The move. STAY will be chosen when the strategy doesn't respond.
181      */
182     public Move decideNextMove() {
183         this.validate();
184         DecisionMaker.LOGGER.trace("Player {} advancing time. ", new Object[]{this.player.getName()});
185         final SessionPseudoClock clock = this.session.getSessionClock();
186         clock.advanceTime(1, TimeUnit.MINUTES);
187         // decide
188         DecisionMaker.LOGGER.trace("Player {} deciding. ", new Object[]{this.player.getName()});
189         this.latestDecision = null;
190         this.session.fireAllRules();
191         // increase turn number
192         final CurrentTurn turn = (CurrentTurn) this.session.getObject(this.currentTurn);
193         this.session.update(this.currentTurn, new CurrentTurn(turn.getNumber() + 1));
194         // store the decision
195         if (this.latestDecision == null) {
196             DecisionMaker.LOGGER.info("Player {} didn't make a decision. STAY forced.", this.player.getName());
197             return Move.STAY;
198         } else {
199             DecisionMaker.LOGGER.info("Player {} final decision is {}. ", this.player.getName(), this.latestDecision);
200             return this.latestDecision;
201         }
202     }

236     public void notifyOfDeath(final PlayerDeathEvent evt) {
237         this.playerEvents.insert(evt);
238         final Player p = evt.getPlayer();
239         // remove player from the WM
240         for (final Map.Entry<Node, FactHandle> entry : this.handles.remove(p).entrySet()) {
241             this.session.retract(entry.getValue());
242         }
243     }
244 
245     public void notifyOfPlayerMove(final PlayerMoveEvent evt) {
246         final Player p = evt.getPlayer();
247         this.playerEvents.insert(evt);
248         // update player positions
249         if (!this.handles.containsKey(p)) {
250             this.handles.put(p, new HashMap<Node, FactHandle>());
251         }
252         final Map<Node, FactHandle> playerHandles = this.handles.get(p);
253         final Set<Node> untraversedNodes = new HashSet<Node>(playerHandles.keySet());
254         for (final Node n : evt.getNodes()) {
255             if (!playerHandles.containsKey(n)) { // worm occupies a new node
256                 final FactHandle fh = this.session.insert(new Worm(p, n));
257                 playerHandles.put(n, fh);
258             }
259             untraversedNodes.remove(n);
260         }
261         for (final Node n : untraversedNodes) { // worm no longer
262                                                 // occupies a node
263             final FactHandle fh = playerHandles.remove(n);
264             this.session.retract(fh);
265         }
266     }

272     @Override
273     public void send(final Object object) {
274         this.validate();
275         if (object instanceof Move) {
276             if (this.latestDecision != null) {
277                 DecisionMaker.LOGGER.debug("Player {} has changed the decision from {} to {}.", new Object[]{
278                         this.player.getName(), this.latestDecision, object});
279             }
280             this.latestDecision = (Move) object;
281         } else {
282             DecisionMaker.LOGGER.warn("Player {} indicated an invalid move {}.", new Object[]{this.player.getName(),
283                     this.latestDecision});
284         }
285     }

287     /**
288      * Clean up after this Drools instance. Will terminate the session and leave
289      * all the objects up for garbage collection. Only call once and then don't
290      * use this object anymore.
291      * 
292      * @return False if already terminated.
293      */
294     public boolean terminate() {
295         if (this.isDisposed) {
296             DecisionMaker.LOGGER.warn("Player {} already terminated.", new Object[]{this.player.getName()});
297             return false;
298         } else {
299             DecisionMaker.LOGGER.info("Terminating player {}.", new Object[]{this.player.getName()});
300             if (this.sessionAudit != null) {
301                 this.sessionAudit.close();
302             }
303             this.halt();
304             this.session.dispose();
305             return true;
306         }
307     }
308 
309     private void validate() {
310         if (this.isDisposed) {
311             throw new IllegalStateException("Player " + this.player.getName() + " already terminated!");
312         }
