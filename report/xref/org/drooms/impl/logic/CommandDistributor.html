35  /**
36   * Receives state changes ({@link Command}s) from the {@link GameController} and
37   * distributes them to all the player strategies ({@link DecisionMaker} to
38   * process them and make {@link Move} decisions on them.
39   */
40  public class CommandDistributor {

42      private static class DecisionMakerUnit implements Callable<Move> {
43  
44          private final DecisionMaker playerLogic;
45          private final List<Command> commands;
46  
47          public DecisionMakerUnit(final DecisionMaker m, final List<Command> commands) {
48              this.playerLogic = m;
49              this.commands = commands;
50          }
51  
52          @Override
53          public Move call() throws Exception {
54              for (final Command command : this.commands) {
55                  command.perform(this.playerLogic);
56              }
57              return this.playerLogic.decideNextMove();
58          }

64      private static Map<Player, Deque<Node>> retrieveNewPlayerPositions(final List<Command> commands) {
65          final Map<Player, Deque<Node>> positions = new HashMap<>();
66          for (final Command command : commands) {
67              if (command instanceof MovePlayerCommand) {
68                  final MovePlayerCommand cmd = (MovePlayerCommand) command;
69                  positions.put(cmd.getPlayer(), cmd.getNodes());
70              }
71          }
72          return Collections.unmodifiableMap(positions);
73      }
74  
75      private static Set<Player> retrievePlayersToRemove(final List<Command> commands) {
76          final Set<Player> players = new HashSet<>();
77          for (final Command command : commands) {
78              if (command instanceof DeactivatePlayerCommand) {
79                  // player being removed from the game
80                  players.add(((PlayerRelated) command).getPlayer());
81              }
82          }
83          return Collections.unmodifiableSet(players);
84      }

95      /**
96       * Initialize the class.
97       * 
98       * @param playground
99       *            The playground on which the game is happening.
100      * @param players
101      *            The players taking part in the game.
102      * @param report
103      *            The game listener.
104      * @param properties
105      *            Configuration of the game.
106      * @param reportFolder
107      *            Where to report to.
108      * @param playerTimeoutInSeconds
109      *            How much time the player strategies should be given to make
110      *            move decisions.
111      */
112     public CommandDistributor(final Playground playground, final Collection<Player> players,
113             final GameProgressListener report, final GameProperties properties, final File reportFolder,
114             final int playerTimeoutInSeconds) {
115         for (final Player player : players) {
116             final PathTracker tracker = new PathTracker(playground, player);
117             this.trackers.put(player, tracker);
118             this.players.put(player, new DecisionMaker(player, tracker, properties, reportFolder));
119         }
120         this.listeners.add(report);
121         this.playerTimeoutInSeconds = playerTimeoutInSeconds;
122     }

131         if (!this.listeners.contains(listener)) {
132             this.listeners.add(listener);
133             return true;

139     /**
140      * Execute the commands.
141      * 
142      * @param commands
143      *            A collection of state changes, to be handed over to strategies
144      *            in this exact order.
145      * @return Strategy decisions.
146      */
147     public Map<Player, Move> execute(final List<Command> commands) {
148         // hint GC to potentially not interrupt decision making later
149         System.gc();
150         CommandDistributor.LOGGER.info("First reporting what happens in this turn.");
151         for (final GameProgressListener listener : this.listeners) {
152             listener.nextTurn();
153         }
154         for (final Command command : commands) {
155             CommandDistributor.LOGGER.info("Will process command: {}", command);
156             for (final GameProgressListener listener : this.listeners) {
157                 command.report(listener);
158             }
159         }
160         CommandDistributor.LOGGER.info("Now passing these changes to players.");
161         final Map<Player, Deque<Node>> positions = CommandDistributor.retrieveNewPlayerPositions(commands);
162         final Set<Player> playersToRemove = CommandDistributor.retrievePlayersToRemove(commands);
163         final Map<Player, Move> moves = new HashMap<Player, Move>();
164         for (final Map.Entry<Player, DecisionMaker> entry : this.players.entrySet()) {
165             final Player player = entry.getKey();
166             if (playersToRemove.contains(player)) {
167                 continue;
168             }
169             final DecisionMaker playerLogic = entry.getValue();
170             this.trackers.get(player).movePlayers(positions);
171             CommandDistributor.LOGGER.debug("Processing player {}.", player.getName());
172             final DecisionMakerUnit dmu = new DecisionMakerUnit(playerLogic, commands);
173             // begin the time-box for a player strategy
174             final Future<Move> move = this.e.submit(dmu);
175             try {
176                 moves.put(player, move.get(this.playerTimeoutInSeconds, TimeUnit.SECONDS));
177             } catch (InterruptedException | ExecutionException e) {
178                 CommandDistributor.LOGGER.warn("Player {} error during decision-making, STAY forced.",
179                         player.getName(), e);
180                 moves.put(player, Move.STAY);
181             } catch (final TimeoutException e) {
182                 CommandDistributor.LOGGER.warn("Player {}, didn't reach a decision in time, STAY forced.",
183                         player.getName());
184                 moves.put(player, Move.STAY);
185             } finally {
186                 move.cancel(true);
187                 playerLogic.halt(); // otherwise other players' are slowed down
188             }
189             // end the time-box for a player strategy
190             CommandDistributor.LOGGER.debug("Player {} processed.", player.getName());
191         }
192         // purge dead players
193         for (final Player p : playersToRemove) {
194             CommandDistributor.LOGGER.debug("Removing player {}.", p.getName());
195             final DecisionMaker dm = this.players.remove(p);
196             dm.terminate();
197             this.trackers.remove(p);
198         }
199         CommandDistributor.LOGGER.info("Turn processed completely.");
200         return Collections.unmodifiableMap(moves);
201     }

203     public GameProgressListener getReport() {
204         return this.listeners.get(0);
205     }

207     /**
208      * Clean up when the game is over. This instance shouldn't be used anymore
209      * after this method is called. Not calling this method after the game may
210      * result in the JVM not terminating, since the executors will still be
211      * active.
212      */
213     public void terminate() {
214         for (final Map.Entry<Player, DecisionMaker> entry : this.players.entrySet()) {
215             entry.getValue().terminate();
216         }
217         this.e.shutdownNow();
218     }
