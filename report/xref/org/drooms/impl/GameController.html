41  /**
42   * Provide a common ground for various types of games. We introduce a couple of
43   * concepts and let the implementing classes specify the rules around those
44   * concepts. The following concepts will be shared by all games extending this
45   * class:
46   * 
47   * <ul>
48   * <li>Each player gets one worm. Properties of these worms come from the game
49   * config and will be explained later. List of players comes from the player
50   * config.</li>
51   * <li>When a worm collides with something, it is terminated. Collisions are
52   * determined by classes extending this one.</li>
53   * <li>When a worm's past couple decisions were all STAY (see {@link Move}), the
54   * worm may be terminated. This is controlled by the classes extending this one.
55   * </li>
56   * <li>At the end of each turn, each non-terminated worm is rewarded for
57   * surviving.</li>
58   * <li>Terminated worms will disappear from the playground in the next turn.</li>
59   * <li>In each turn, a collectible item of a certain value may appear in the
60   * playground. These collectibles will disappear after a certain amount of
61   * turns. Worms who collect them in the meantime will be rewarded. How often,
62   * how valuable and how persistent the collectibles are, that depends on the
63   * classes extending this one.</li>
64   * <li>Upon collecting an item, the worm's length will increase by 1.</li>
65   * <li>Game ends either when there are between 0 and 1 worms standing or when a
66   * maximum number of turns is reached.</li>
67   * <li>At the end of the game, a player whose worm has the most points is
68   * declared the winner.</li>
69   * </ul>
70   * 
71   * <p>
72   * Some of the decisions can be made by classes extending this one. These are
73   * clearly described above.
74   * </p>
75   * 
76   * <p>
77   * This class depends on a couple properties from the game config:
78   * </p>
79   * 
80   * <dl>
81   * <dt>playground.file</dt>
82   * <dd>From where the {@link Playground} should be read. Will be passed to
83   * {@link DefaultPlayground#read(InputStream)}.</dd>
84   * <dt>worm.length.start (defaults to 3)</dt>
85   * <dd>Length of the worm at the start of the game. Actually, initially each
86   * worm will only have a length of 1. But as it first moves its head, the tail
87   * of the worm will stay where the head was, until the worm reaches the
88   * specified starting length. From then on, the length will be kept constant and
89   * only changed upon collecting an item.</dd>
90   * <dt>worm.max.turns (defaults to 1000)</dt>
91   * <dd>Maximum length of the game, in case more than 1 worm keeps on surviving.</dd>
92   * <dt>worm.max.inactive.turns (defaults to 3)</dt>
93   * <dd>Maximum number of turns of inactivity after which a player may be
94   * terminated, if the game decides so.</dd>
95   * <dt>worm.timeout.seconds (defaults to 1)</dt>
96   * <dd>The maximum amount of time that the {@link Player}'s {@link Strategy} has
97   * to make a decision on the next movement of the worm. If it doesn't make it in
98   * time, STAY is enforced, potentially leading to the worm being terminated for
99   * inactivity.</dd>
100  * <dt>worm.survival.bonus (defaults to 1)</dt>
101  * <dd>The amount of points that the worm will be awarded upon surviving another
102  * turn.</dd>
103  * </dl>
104  */

187     @Override
188     public GameProgressListener play(final String id,
189             final Properties gameConfig, final Properties playerConfig) {
190         // prepare the playground
191         DefaultPlayground playground;
192         try (final InputStream fis = new FileInputStream(new File(
193                 gameConfig.getProperty("playground.file")))) {
194             playground = DefaultPlayground.read(fis);
195         } catch (final IOException e) {
196             throw new IllegalArgumentException(
197                     "Playground file cannot be read!", e);
198         }
199         final int wormLength = Integer.valueOf(gameConfig.getProperty(
200                 "worm.length.start", "3"));
201         final int allowedInactiveTurns = Integer.valueOf(gameConfig
202                 .getProperty("worm.max.inactive.turns", "3"));
203         final int allowedTurns = Integer.valueOf(gameConfig.getProperty(
204                 "worm.max.turns", "1000"));
205         final int wormSurvivalBonus = Integer.valueOf(gameConfig.getProperty(
206                 "worm.survival.bonus", "1"));
207         final int wormTimeout = Integer.valueOf(gameConfig.getProperty(
208                 "worm.timeout.seconds", "1"));
209         // prepare players and their starting positions
210         final List<Player> players = new PlayerAssembly(playerConfig)
211                 .assemblePlayers();
212         final List<Node> startingPositions = playground.getStartingPositions();
213         final int playersSupported = startingPositions.size();
214         final int playersAvailable = players.size();
215         if (playersSupported < playersAvailable) {
216             throw new IllegalArgumentException(
217                     "The playground doesn't support " + playersAvailable
218                             + " players, only " + playersSupported + "! ");
219         }
220         int i = 0;
221         for (final Player player : players) {
222             final Deque<Node> pos = new LinkedList<Node>();
223             pos.push(startingPositions.get(i));
224             this.setPlayerPosition(player, pos);
225             this.setPlayerLength(player, wormLength);
226             GameController.LOGGER.info("Player {} assigned position {}.",
227                     player.getName(), i);
228             i++;
229         }
230         // prepare situation
231         final CommandDistributor playerControl = new CommandDistributor(
232                 playground, players, new XmlProgressListener(playground,
233                         gameConfig, id), wormTimeout);
234         final Set<Player> currentPlayers = new HashSet<Player>(players);
235         Map<Player, Move> decisions = new HashMap<Player, Move>();
236         for (final Player p : currentPlayers) { // initialize players
237             decisions.put(p, Move.STAY);
238         }
239         // start the game
240         int turnNumber = 0;
241         do {
242             GameController.LOGGER.info("--- Starting turn no. {}.", turnNumber);
243             final List<Command> commands = new LinkedList<>();
244             // remove inactive worms
245             for (final Player player : this.performInactivityDetection(
246                     currentPlayers, turnNumber, allowedInactiveTurns)) {
247                 currentPlayers.remove(player);
248                 commands.add(new DeactivatePlayerCommand(player));
249             }
250             // move the worms
251             for (final Player p : currentPlayers) {
252                 final Move m = decisions.get(p);
253                 this.addDecision(p, m, turnNumber);
254                 final Deque<Node> newPosition = this.performPlayerMove(p, m);
255                 this.setPlayerPosition(p, newPosition);
256                 commands.add(new MovePlayerCommand(p, m, newPosition));
257             }
258             // resolve worms colliding
259             for (final Player player : this.performCollisionDetection(
260                     playground, currentPlayers)) {
261                 currentPlayers.remove(player);
262                 commands.add(new CrashPlayerCommand(player));
263             }
264             if (turnNumber > 0) {
265                 // reward surviving worms; but not in the first round
266                 for (final Player p : currentPlayers) {
267                     this.reward(p, 1);
268                     commands.add(new RewardSurvivalCommand(p, wormSurvivalBonus));
269                 }
270             }
271             // expire uncollected collectibles
272             final Set<Collectible> removeCollectibles = new HashSet<Collectible>();
273             for (final Collectible c : this.collectiblesByNode.values()) {
274                 if (c.expires() && turnNumber >= c.expiresInTurn()) {
275                     removeCollectibles.add(c);
276                 }
277             }
278             for (final Collectible c : removeCollectibles) {
279                 commands.add(new RemoveCollectibleCommand(c, this.getNode(c)));
280                 this.removeCollectible(c);
281             }
282             // add points for collected collectibles
283             for (final Map.Entry<Collectible, Player> entry : this
284                     .performCollectibleCollection(currentPlayers).entrySet()) {
285                 final Collectible c = entry.getKey();
286                 final Player p = entry.getValue();
287                 this.reward(p, c.getPoints());
288                 commands.add(new CollectCollectibleCommand(c, p, this
289                         .getNode(c)));
290                 this.removeCollectible(c);
291                 this.setPlayerLength(p, this.getPlayerLength(p) + 1);
292             }
293             // distribute new collectibles
294             for (final Map.Entry<Collectible, Node> entry : this
295                     .performCollectibleDistribution(gameConfig, playground,
296                             currentPlayers, turnNumber).entrySet()) {
297                 final Collectible c = entry.getKey();
298                 final Node n = entry.getValue();
299                 this.addCollectible(c, n);
300                 commands.add(new AddCollectibleCommand(c, n));
301             }
302             // make the move decision
303             decisions = playerControl.execute(commands);
304             turnNumber++;
305             if (turnNumber == allowedTurns) {
306                 GameController.LOGGER
307                         .info("Reached a pre-defined limit of {} turns. Terminating game.",
308                                 allowedTurns);
309                 break;
310             } else if (currentPlayers.size() < 2) {
311                 GameController.LOGGER
312                         .info("There are no more players. Terminating game.");
313                 break;
314             }
315         } while (true);
316         playerControl.terminate(); // clean up all the sessions
317         // output player status
318         GameController.LOGGER.info("--- Game over.");
319         for (final Map.Entry<Player, Integer> entry : this.playerPoints
320                 .entrySet()) {
321             GameController.LOGGER.info("Player {} earned {} points.", entry
322                     .getKey().getName(), entry.getValue());
323         }
324         return playerControl.getReport();
325     }
