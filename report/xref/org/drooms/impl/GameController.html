42  /**
43   * Provide a common ground for various types of games. We introduce a couple of
44   * concepts and let the implementing classes specify the rules around those
45   * concepts. The following concepts will be shared by all games extending this
46   * class:
47   * 
48   * <ul>
49   * <li>Each player gets one worm. Properties of these worms come from the game config and will be explained later. List
50   * of players comes from the player config.</li>
51   * <li>When a worm collides with something, it is terminated. Collisions are determined by classes extending this one.</li>
52   * <li>When a worm's past couple decisions were all STAY (see {@link Move}), the worm may be terminated. This is
53   * controlled by the classes extending this one.</li>
54   * <li>When a turn ends, worms may be rewarded for surviving. How and when, that depends on the classes extending this
55   * one.</li>
56   * <li>Terminated worms will disappear from the playground in the next turn.</li>
57   * <li>In each turn, a collectible item of a certain value may appear in the playground. These collectibles will
58   * disappear after a certain amount of turns. Worms who collect them in the meantime will be rewarded. How often, how
59   * valuable and how persistent the collectibles are, that depends on the classes extending this one.</li>
60   * <li>Upon collecting an item, the worm's length will increase by 1.</li>
61   * <li>Game ends either when there are between 0 and 1 worms standing or when a maximum number of turns is reached.</li>
62   * <li>At the end of the game, a player whose worm has the most points is declared the winner.</li>
63   * </ul>
64   * 
65   * <p>
66   * Some of the decisions can be made by classes extending this one. These are clearly described above. This class
67   * depends on properties as defined in {@link GameProperties}.
68   * </p>
69   * 
70   */

223     @Override
224     public Map<Player, Integer> play(final Playground playground, final Collection<Player> players,
225             final File reportFolder) {
226         if (this.gameConfig == null) {
227             throw new IllegalStateException("Game context had not been set!");
228         }
229         // make sure a game isn't played more than once
230         if (this.played.get()) {
231             throw new IllegalStateException("This game had already been played.");
232         }
233         this.played.set(true);
234         // prepare the playground
235         final int wormLength = this.gameConfig.getStartingWormLength();
236         final int allowedInactiveTurns = this.gameConfig.getMaximumInactiveTurns();
237         final int allowedTurns = this.gameConfig.getMaximumTurns();
238         final int wormSurvivalBonus = this.gameConfig.getDeadWormBonus();
239         final int wormTimeout = this.gameConfig.getStrategyTimeoutInSeconds();
240         // prepare players and their starting positions
241         final List<Node> startingPositions = playground.getStartingPositions();
242         final int playersSupported = startingPositions.size();
243         final int playersAvailable = players.size();
244         if (playersSupported < playersAvailable) {
245             throw new IllegalArgumentException("The playground doesn't support " + playersAvailable + " players, only "
246                     + playersSupported + "! ");
247         }
248         int i = 0;
249         for (final Player player : players) {
250             final Deque<Node> pos = new LinkedList<Node>();
251             pos.push(startingPositions.get(i));
252             this.setPlayerPosition(player, pos);
253             this.setPlayerLength(player, wormLength);
254             GameController.LOGGER.info("Player {} assigned position {}.", player.getName(), i);
255             i++;
256         }
257         // prepare situation
258         this.reporter = new XmlProgressListener(playground, players, this.gameConfig);
259         final CommandDistributor playerControl = new CommandDistributor(playground, players, this.reporter,
260                 this.gameConfig, reportFolder, wormTimeout);
261         for (final GameProgressListener listener : this.listeners) {
262             playerControl.addListener(listener);
263         }
264         final Set<Player> currentPlayers = new HashSet<Player>(players);
265         Map<Player, Move> decisions = new HashMap<Player, Move>();
266         for (final Player p : currentPlayers) { // initialize players
267             decisions.put(p, Move.STAY);
268         }
269         // start the game
270         int turnNumber = 0;
271         do {
272             GameController.LOGGER.info("--- Starting turn no. {}.", turnNumber);
273             final int preRemoval = currentPlayers.size();
274             final List<Command> commands = new LinkedList<>();
275             // remove inactive worms
276             for (final Player player : this
277                     .performInactivityDetection(currentPlayers, turnNumber, allowedInactiveTurns)) {
278                 currentPlayers.remove(player);
279                 commands.add(new DeactivatePlayerCommand(player));
280                 GameController.LOGGER.info("Player {} will be removed for inactivity.", player.getName());
281             }
282             // move the worms
283             for (final Player p : currentPlayers) {
284                 final Move m = decisions.get(p);
285                 this.addDecision(p, m, turnNumber);
286                 final Deque<Node> newPosition = this.performPlayerMove(p, m);
287                 this.setPlayerPosition(p, newPosition);
288                 commands.add(new MovePlayerCommand(p, m, newPosition));
289             }
290             // resolve worms colliding
291             for (final Player player : this.performCollisionDetection(playground, currentPlayers)) {
292                 currentPlayers.remove(player);
293                 commands.add(new CrashPlayerCommand(player));
294             }
295             final int postRemoval = currentPlayers.size();
296             for (final Map.Entry<Player, Integer> entry : this.performSurvivalRewarding(players, currentPlayers,
297                     preRemoval - postRemoval, wormSurvivalBonus).entrySet()) {
298                 final Player p = entry.getKey();
299                 final int amount = entry.getValue();
300                 this.reward(p, amount);
301                 commands.add(new RewardSurvivalCommand(p, amount));
302             }
303             // expire uncollected collectibles
304             final Set<Collectible> removeCollectibles = new HashSet<Collectible>();
305             for (final Collectible c : this.collectiblesByNode.values()) {
306                 if (c.expires() && turnNumber >= c.expiresInTurn()) {
307                     removeCollectibles.add(c);
308                 }
309             }
310             for (final Collectible c : removeCollectibles) {
311                 commands.add(new RemoveCollectibleCommand(c));
312                 this.removeCollectible(c);
313             }
314             // add points for collected collectibles
315             for (final Map.Entry<Collectible, Player> entry : this.performCollectibleCollection(currentPlayers)
316                     .entrySet()) {
317                 final Collectible c = entry.getKey();
318                 final Player p = entry.getValue();
319                 this.reward(p, c.getPoints());
320                 commands.add(new CollectCollectibleCommand(c, p));
321                 this.removeCollectible(c);
322                 this.setPlayerLength(p, this.getPlayerLength(p) + 1);
323             }
324             // distribute new collectibles
325             for (final Collectible c : this.performCollectibleDistribution(this.gameConfig, playground, currentPlayers, turnNumber)) {
326                 this.addCollectible(c);
327                 commands.add(new AddCollectibleCommand(c));
328             }
329             // make the move decision
330             decisions = playerControl.execute(commands);
331             turnNumber++;
332             if (turnNumber == allowedTurns) {
333                 GameController.LOGGER.info("Reached a pre-defined limit of {} turns. Terminating game.", allowedTurns);
334                 break;
335             } else if (currentPlayers.size() < 2) {
336                 GameController.LOGGER.info("There are no more players. Terminating game.");
337                 break;
338             }
339         } while (true);
340         playerControl.terminate(); // clean up all the sessions
341         // output player status
342         GameController.LOGGER.info("--- Game over.");
343         for (final Map.Entry<Player, Integer> entry : this.playerPoints.entrySet()) {
344             GameController.LOGGER.info("Player {} earned {} points.", entry.getKey().getName(), entry.getValue());
345         }
346         return Collections.unmodifiableMap(this.playerPoints);
347     }

385     /**
386      * Build the playground from an input stream. Each line in that stream
387      * represents one row on the playground. Each '#' in that line represents a
388      * wall node, ' ' represents a node where the worm can move, '@' represents
389      * a possible starting position for a worm. (Starting positions also can be
390      * moved into.) Any other sign, other than a line break, will result in an
391      * exception.
392      * 
393      * @param name
394      *            Name for the new playground.
395      * @param source
396      *            Stream in question.
397      * @return Playground constructed from that stream.
398      */
399     @Override
400     public Playground buildPlayground(final String name, final InputStream source) {
401         try {
402             final List<String> lines = IOUtils.readLines(source);
403             Collections.reverse(lines); // this way, 0,0 is bottom left
404             return new DefaultPlayground(name, lines);
405         } catch (final Exception ex) {
406             throw new IllegalStateException("Cannot read playground " + name, ex);
407         }
408     }
