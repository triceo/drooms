15  /**
16   * Command-line interface for the application. It enforces following options on
17   * the command line:
18   * 
19   * <dl>
20   * <dt>-s &lt;scenario&gt;</dt>
21   * <dd>Provides a {@link Playground} description on which the game is to be played out..</dd>
22   * <dt>-p &lt;file&gt;</dt>
23   * <dd>Provides a player configuration file, as described in
24   * {@link GameController#play(Playground, java.util.Collection, File)}.</dd>
25   * <dt>-g &lt;file&gt;</dt>
26   * <dd>Provides a game configuration file, as described in
27   * {@link GameController#play(Playground, java.util.Collection, File)}.</dd>
28   * </dl>
29   * 
30   * Not providing any of those or pointing to unreadable (non-existent) files
31   * should result in a help message being printed out and the application being
32   * terminated.
33   */
34  public class CLI {
35  
36      private static final CLI INSTANCE = new CLI();
37  
38      /**
39       * Return the single instance of this class.
40       * 
41       * @return The instance.
42       */
43      public static CLI getInstance() {
44          return CLI.INSTANCE;
45      }
46  
47      private final Options options = new Options();
48  
49      private final Option reports = new Option("r", "reports", true, "A folder to store reports in.");
50      private final Option playground = new Option("s", "scenario", true, "A path to the playground config file.");
51      private final Option players = new Option("p", "players", true, "A path to the player config file.");
52      private final Option game = new Option("g", "game", true, "A path to the game config file.");
53  
54      private String errorMessage = null;
55      private boolean isError = false;
56  
57      /**
58       * The constructor is hidden, as should be with the singleton pattern.
59       */
60      private CLI() {
61          this.options.addOption(this.reports);
62          this.playground.setRequired(true);
63          this.options.addOption(this.playground);
64          this.game.setRequired(true);
65          this.options.addOption(this.game);
66          this.players.setRequired(true);
67          this.options.addOption(this.players);
68      }
69  
70      /**
71       * Prints a help message, describing the usage of the app from the
72       * command-line.
73       */
74      public void printHelp() {
75          if (this.isError) {
76              System.out.println(this.errorMessage);
77          }
78          final HelpFormatter formatter = new HelpFormatter();
79          formatter.printHelp("java -jar drooms.jar", this.options, true);
80      }
81  
82      /**
83       * Process the command-line arguments.
84       * 
85       * @param args
86       *            The arguments.
87       * @return Files passed from the command line. First is the scenario, second
88       *         is the game config, third is the player config, fourth
89       *         (optionally) the directory in which to store reports.
90       */
91      public File[] process(final String[] args) {
92          this.isError = false;
93          final CommandLineParser parser = new GnuParser();
94          try {
95              final CommandLine cli = parser.parse(this.options, args);
96              final File scenario = new File(cli.getOptionValue(this.playground.getOpt()));
97              if (!scenario.exists() || !scenario.canRead()) {
98                  this.setError("Provided scenario file cannot be read!");
99                  return null;
100             }
101             final File gameConfig = new File(cli.getOptionValue(this.game.getOpt()));
102             if (!gameConfig.exists() || !gameConfig.canRead()) {
103                 this.setError("Provided game config file cannot be read!");
104                 return null;
105             }
106             final File playerConfig = new File(cli.getOptionValue(this.players.getOpt()));
107             if (!playerConfig.exists() || !playerConfig.canRead()) {
108                 this.setError("Provided player config file cannot be read!");
109                 return null;
110             }
111             final String reports = cli.getOptionValue(this.reports.getOpt());
112             if (reports == null) {
113                 return new File[]{scenario, gameConfig, playerConfig};
114             } else {
115                 final File reportsDir = new File(reports + File.separator);
116                 reportsDir.mkdirs();
117                 return new File[]{scenario, gameConfig, playerConfig, reportsDir};
118             }
119         } catch (final ParseException e) {
120             this.setError(e.getMessage());
121             return null;
122         }
123     }
124 
125     private boolean setError(final String message) {
126         if (!this.isError) {
127             this.isError = true;
128             this.errorMessage = message;
129             return true;
130         }
131         return false;
132     }
