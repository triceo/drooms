3   /**
4    * Represents an item that occasionally pops up on the {@link Playground}. When
5    * a worm's head reaches this item, the {@link Player} in question will be
6    * rewarded by a specific amount of points.
7    * 
8    * No two instances ever {@link #equals(Object)}.
9    */
10  public class Collectible {
11  
12      private final int expiresInTurn, points;
13  
14      /**
15       * Construct an item that never expires.
16       * 
17       * @param points
18       *            How many points to award. Must be > 0.
19       */
20      public Collectible(final int points) {
21          if (points <= 0) {
22              throw new IllegalArgumentException(
23                      "Collectible must have a positive amount of points.");
24          }
25          this.expiresInTurn = -1;
26          this.points = points;
27      }
28  
29      /**
30       * Construct the item.
31       * 
32       * @param points
33       *            How many points to award. Must be > 0.
34       * @param expiresInTurn
35       *            At which turn the item expires. Must be > 0.
36       */
37      public Collectible(final int points, final int expiresInTurn) {
38          if (expiresInTurn <= 0) {
39              throw new IllegalArgumentException(
40                      "Expiration must be a positive number.");
41          } else if (points <= 0) {
42              throw new IllegalArgumentException(
43                      "Collectible must have a positive amount of points.");
44          }
45          this.expiresInTurn = expiresInTurn;
46          this.points = points;
47      }
48  
49      /**
50       * Whether or not this item will disappear from the {@link Playground} at
51       * some point in time.
52       * 
53       * @return True if the item will disappear, false if it stays indefinitely.
54       */
55      public boolean expires() {
56          return (this.expiresInTurn >= 0);
57      }
58  
59      /**
60       * The number of the turn at the beginning of which this item will disappear
61       * from the {@link Playground}.
62       * 
63       * @return A number >= 0 when {@link #expires()} is true, -1 otherwise.
64       */
65      public int expiresInTurn() {
66          return this.expiresInTurn;
67      }
68  
69      /**
70       * Number of points to award to {@link Player} in case a worm successfully
71       * reaches the item.
72       * 
73       * @return A positive number.
74       */
75      public int getPoints() {
76          return this.points;
77      }
78  
79      @Override

81          final StringBuilder builder = new StringBuilder();
82          builder.append("Collectible [expiresInTurn=")
