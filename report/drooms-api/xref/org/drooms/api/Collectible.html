3   /**
4    * Represents an item that occasionally pops up on the {@link Playground}. When
5    * a worm's head reaches this item, the {@link Player} in question will be
6    * rewarded by a specific amount of points.
7    * 
8    * No two instances ever {@link #equals(Object)}.
9    */
10  public class Collectible {
11  
12      private final int expiresInTurn, points;
13      private final Node at;

15      
16      public Node getAt() {
17          return at;
18      }

20      /**
21       * Construct an item that never expires.
22       * 
23       * @param n
24       *            Where the collectible is located.
25       * @param points
26       *            How many points to award. Must be > 0.
27       */
28      public Collectible(final Node n, final int points) {
29          if (points <= 0) {
30              throw new IllegalArgumentException(
31                      "Collectible must have a positive amount of points.");
32          }
33          this.expiresInTurn = -1;
34          this.points = points;
35          this.at = n;
36      }

38      /**
39       * Construct the item.
40       * 
41       * @param n
42       *            Where the collectible is located.
43       * @param points
44       *            How many points to award. Must be > 0.
45       * @param expiresInTurn
46       *            At which turn the item expires. Must be > 0.
47       */
48      public Collectible(final Node n, final int points, final int expiresInTurn) {
49          if (expiresInTurn <= 0) {
50              throw new IllegalArgumentException(
51                      "Expiration must be a positive number.");
52          } else if (points <= 0) {
53              throw new IllegalArgumentException(
54                      "Collectible must have a positive amount of points.");
55          }
56          this.expiresInTurn = expiresInTurn;
57          this.points = points;
58          this.at = n;
59      }
60  
61      /**
62       * Whether or not this item will disappear from the {@link Playground} at
63       * some point in time.
64       * 
65       * @return True if the item will disappear, false if it stays indefinitely.
66       */
67      public boolean expires() {
68          return (this.expiresInTurn >= 0);
69      }
70  
71      /**
72       * The number of the turn at the beginning of which this item will disappear
73       * from the {@link Playground}.
74       * 
75       * @return A number >= 0 when {@link #expires()} is true, -1 otherwise.
76       */
77      public int expiresInTurn() {
78          return this.expiresInTurn;
79      }
80  
81      /**
82       * Number of points to award to {@link Player} in case a worm successfully

84       * 
85       * @return A positive number.
86       */
87      public int getPoints() {

89      }
