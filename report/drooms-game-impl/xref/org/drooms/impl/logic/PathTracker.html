25   * A helper class for the strategies to be able to quickly and easily find paths
26   * from one {@link Node} to another.

30      private static UndirectedGraph<Node, Edge> cloneGraph(final Graph<Node, Edge> src,
31              final Collection<Node> removeNodes) {
32          final UndirectedGraph<Node, Edge> clone = new UndirectedSparseGraph<>();
33          for (final Edge e : src.getEdges()) {
34              final boolean isEdgeAdded = clone.addEdge(e, src.getIncidentVertices(e));
35              if (!isEdgeAdded) {
36                  throw new IllegalStateException("Failed cloning graph. This surely is a bug in Drooms.");
37              }
38          }
39          for (final Node node : removeNodes) {
40              clone.removeVertex(node);
41          }
42          return Graphs.unmodifiableUndirectedGraph(clone);
43      }

90      /**
91       * Find the shortest path from the start node that leads through the other
92       * two nodes in any order.
93       * 
94       * @param start
95       *            Beginning of the path.
96       * @param node2
97       *            Any of the other two nodes.
98       * @param node3
99       *            Any of the other two nodes.
100      * @return Unmodifiable list of nodes on the path, ordered from start to
101      *         end. Empty if path cannot be found.
102      */
103     public List<Edge> getPath(final Node start, final Node node2, final Node node3) {
104         final List<Edge> path2 = this.getPath(start, node2);
105         final List<Edge> path3 = this.getPath(start, node3);
106         final List<Edge> path23 = new ArrayList<Edge>(this.getPath(node2, node3));
107         final List<Edge> result = new ArrayList<Edge>();
108         if (path2.size() > path3.size()) {
109             result.addAll(path3);
110             Collections.reverse(path23);
111         } else {
112             result.addAll(path2);
113         }
114         result.addAll(path23);
115         return Collections.unmodifiableList(result);
116     }

126     /**
127      * Update the internal state of this class so that future paths can avoid
128      * places where the worms currently reside.
129      * 
130      * @param newPositions
131      *            New current positions of all the worms.
132      */
133     protected void movePlayers(final Map<Player, Deque<Node>> newPositions) {
134         final Set<Node> unavailable = new HashSet<>();
135         for (final Map.Entry<Player, Deque<Node>> entry : newPositions.entrySet()) {
136             Deque<Node> playerNodes = entry.getValue(); 
137             unavailable.addAll(playerNodes);
138             if (entry.getKey() == this.player) {
139                 /*
140                  * the head node needs to remain, since otherwise there would be
141                  * no path between the current position and any other position.
142                  */
143                 unavailable.remove(playerNodes.getFirst());
144             }
145         }
146         final Graph<Node, Edge> graphWithoutPlayers = this.playground.getGraph();
147         this.currentGraph = PathTracker.cloneGraph(graphWithoutPlayers, unavailable);
148         this.currentPath = this.player.getShortestPathAlgorithm(this.currentGraph);
149         this.currentPosition = newPositions.get(this.player).getFirst();
150     }
