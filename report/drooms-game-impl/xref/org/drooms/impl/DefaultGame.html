22  /**
23   * On top of the rules implemented by {@link GameController}, this game
24   * implementation also puts forward some of its own. Those are:
25   * 
26   * <dl>
27   * <dt>Collision detection</dt>
28   * <dd>When a worm reaches a node at the same time as another worm, both are
29   * terminated. When a worm moves into a body of another worm or into a wall,
30   * only this worm is terminated. Specific probabilities and values come from the
31   * game config.</dd>
32   * <dt>Various types of collectibles</dt>
33   * <dd>This class implements three types of collectibles with varying
34   * probabilities of appearance, expirations and valuations. There are cheap ones
35   * that occur all the time, good ones that occur sometimes and extremely
36   * lucrative ones that occur scarcely and don't last long.</dd>
37   * <dt>Simultaneos collections</dt>
38   * <dd>When two worms collect the same item at the same time, it is considered a
39   * collision. Both worms are terminated and neither is awarded points for the
40   * item.</dd>
41   * <dt>Survival bonuses</dt>
42   * <dd>In the turn when at least one worm is removed from the game, either due
43   * to crashing or inactivity, every surviving worm is rewarded. The amount of
44   * the reward is "the number of worms gone so far" multiplied by the bonus
45   * factor.</dd>
46   * <dt>Inactivity enforcement</dt>
47   * <dd>This implementation will terminate worms for inactivity, as described in
48   * the super class.</dd>
49   * </dl>
50   */

53      @Override
54      protected Map<Collectible, Player> performCollectibleCollection(final Collection<Player> players) {
55          final Map<Collectible, Player> collections = new HashMap<Collectible, Player>();
56          for (final Player p : players) {
57              final Node headPosition = this.getPlayerPosition(p).getFirst();
58              final Collectible c = this.getCollectible(headPosition);
59              if (c != null) { // successfully collected
60                  collections.put(c, p);
61              }
62          }
63          return Collections.unmodifiableMap(collections);
64      }

66      @Override
67      protected Map<Collectible, Node> performCollectibleDistribution(final GameProperties gameConfig,
68              final Playground playground, final Collection<Player> players, final int currentTurnNumber) {
69          final Map<Collectible, Node> collectibles = new HashMap<Collectible, Node>();
70          for (final CollectibleType ct : gameConfig.getCollectibleTypes()) {
71              final BigDecimal probability = ct.getProbabilityOfAppearance();
72              final BigDecimal chosen = BigDecimal.valueOf(GameController.RANDOM.nextDouble());
73              if (probability.compareTo(chosen) > 0) {
74                  final double expirationAdjustmentRate = GameController.RANDOM.nextDouble() + 0.5;
75                  final double turnsToLast = expirationAdjustmentRate * ct.getExpiration();
76                  final int expiresIn = (int) Math.round(currentTurnNumber + turnsToLast);
77                  final int points = ct.getPoints();
78                  final Collectible c = new Collectible(points, expiresIn);
79                  collectibles.put(c, this.pickRandomUnusedNode(playground, players));
80              }
81          }
82          return Collections.unmodifiableMap(collectibles);
83      }

85      @Override
86      protected Set<Player> performCollisionDetection(final Playground playground, final Collection<Player> currentPlayers) {
87          final Set<Player> collisions = new HashSet<Player>();
88          for (final Player p1 : currentPlayers) {
89              final Deque<Node> position = this.getPlayerPosition(p1);
90              final Node firstPosition = position.getFirst();
91              if (!playground.isAvailable(firstPosition.getX(), firstPosition.getY())) {
92                  collisions.add(p1);
93                  continue;
94              } else {
95                  // make sure the worm didn't crash into itself
96                  final Set<Node> nodes = new HashSet<Node>(position);
97                  if (nodes.size() < position.size()) {
98                      // a worm occupies one node twice = a crash into itself
99                      collisions.add(p1);
100                 }
101             }
102             for (final Player p2 : currentPlayers) {
103                 if (p1 == p2) {
104                     // the same worm
105                     continue;
106                 }
107                 final Node secondPosition = this.getPlayerPosition(p2).getFirst();
108                 if (firstPosition.equals(secondPosition)) {
109                     // head-on-head collision
110                     collisions.add(p1);
111                     collisions.add(p2);
112                 } else if (position.contains(secondPosition)) {
113                     // head-on-body collision
114                     collisions.add(p2);
115                 }
116             }
117         }
118         return Collections.unmodifiableSet(collisions);
119     }

121     @Override
122     protected Set<Player> performInactivityDetection(final Collection<Player> currentPlayers,
123             final int currentTurnNumber, final int allowedInactiveTurns) {
124         final Set<Player> inactiveWorms = new HashSet<Player>();
125         if (currentTurnNumber > allowedInactiveTurns) {
126             for (final Player p : currentPlayers) {
127                 final List<Move> allMoves = this.getDecisionRecord(p);
128                 final int size = allMoves.size();
129                 final List<Move> relevantMoves = allMoves.subList(Math.max(0, size - allowedInactiveTurns - 1), size);
130                 if (!relevantMoves.contains(Move.STAY)) {
131                     continue;
132                 }
133                 final Set<Move> uniqueMoves = new HashSet<Move>(relevantMoves);
134                 if (uniqueMoves.size() == 1) {
135                     inactiveWorms.add(p);
136                 }
137             }
138         }
139         return Collections.unmodifiableSet(inactiveWorms);
140     }

142     @Override
143     protected Deque<Node> performPlayerMove(final Player player, final Move decision) {
144         // move the head of the worm
145         final Deque<Node> currentPos = this.getPlayerPosition(player);
146         final Node currentHeadPos = currentPos.getFirst();
147         Node newHeadPos;
148         switch (decision) {
149             case UP:
150                 newHeadPos = Node.getNode(currentHeadPos.getX(), currentHeadPos.getY() + 1);
151                 break;
152             case DOWN:
153                 newHeadPos = Node.getNode(currentHeadPos.getX(), currentHeadPos.getY() - 1);
154                 break;
155             case LEFT:
156                 newHeadPos = Node.getNode(currentHeadPos.getX() - 1, currentHeadPos.getY());
157                 break;
158             case RIGHT:
159                 newHeadPos = Node.getNode(currentHeadPos.getX() + 1, currentHeadPos.getY());
160                 break;
161             case STAY:
162                 newHeadPos = currentHeadPos;
163                 break;
164             default:
165                 throw new IllegalStateException("Unknown move!");
166         }
167         // move the head of the snake
168         final Deque<Node> newPosition = new LinkedList<Node>(currentPos);
169         if (decision != Move.STAY) {
170             newPosition.push(newHeadPos);
171         }
172         // make sure the snake is as long as it should be
173         while (newPosition.size() > this.getPlayerLength(player)) {
174             newPosition.removeLast();
175         }
176         // notify
177         return newPosition;
178     }

180     @Override
181     protected Map<Player, Integer> performSurvivalRewarding(final Collection<Player> allPlayers,
182             final Collection<Player> survivingPlayers, final int removedInThisRound, final int rewardAmount) {
183         if (removedInThisRound < 1) {
184             return Collections.emptyMap();
185         }
186         final int amount = rewardAmount * (allPlayers.size() - survivingPlayers.size());
187         final Map<Player, Integer> result = new HashMap<>();
188         for (final Player p : survivingPlayers) {
189             result.put(p, amount);
190         }
191         return Collections.unmodifiableMap(result);
192     }

194     private Node pickRandomUnusedNode(final Playground p, final Collection<Player> players) {
195         final List<Node> nodes = new LinkedList<Node>();
196         // locate available nodes
197         for (int x = 0; x < p.getWidth(); x++) {
198             for (int y = 0; y < p.getHeight(); y++) {
199                 if (p.isAvailable(x, y)) {
200                     nodes.add(Node.getNode(x, y));
201                 }
202             }
203         }
204         // exclude nodes where worms are
205         for (final Player player : players) {
206             nodes.removeAll(this.getPlayerPosition(player));
207         }
208         // exclude nodes where collectibles are
209         final List<Node> nodesCopy = new LinkedList<Node>(nodes);
210         for (final Node n : nodesCopy) {
211             if (this.getCollectible(n) != null) {
212                 nodes.remove(n);
213             }
214         }
215         if (nodes.size() == 0) {
216             return null;
217         } else {
218             return nodes.get(GameController.RANDOM.nextInt(nodes.size()));
219         }
